Criar tabelas no banco Execute uma vez
Copie o SQL abaixo, vá em SQL Editor no painel do Supabase, cole e execute. 
-- Tabelas principais
CREATE TABLE IF NOT EXISTS aulas (
  id BIGSERIAL PRIMARY KEY, numero INTEGER NOT NULL, titulo TEXT NOT NULL,
  nivel TEXT DEFAULT 'A1', tema TEXT, verbos TEXT[], observacoes TEXT,
  video_url TEXT, imagem_url TEXT, criado_em TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS vocabulario (
  id BIGSERIAL PRIMARY KEY, aula_id BIGINT REFERENCES aulas(id) ON DELETE CASCADE,
  espanhol TEXT NOT NULL, portugues TEXT NOT NULL, categoria TEXT DEFAULT 'geral',
  exemplo TEXT, imagem_url TEXT, criado_em TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS frases (
  id BIGSERIAL PRIMARY KEY, aula_id BIGINT REFERENCES aulas(id) ON DELETE CASCADE,
  espanhol TEXT NOT NULL, portugues TEXT NOT NULL, categoria TEXT DEFAULT 'geral',
  imagem_url TEXT, criado_em TIMESTAMPTZ DEFAULT NOW()
);
-- Alunos
CREATE TABLE IF NOT EXISTS alunos (
  id BIGSERIAL PRIMARY KEY, nome TEXT NOT NULL, email TEXT NOT NULL UNIQUE,
  codigo TEXT NOT NULL UNIQUE, ativo BOOLEAN DEFAULT TRUE, criado_em TIMESTAMPTZ DEFAULT NOW()
);
-- Progresso
CREATE TABLE IF NOT EXISTS progresso (
  id BIGSERIAL PRIMARY KEY, aluno_id BIGINT REFERENCES alunos(id) ON DELETE CASCADE,
  aula_id BIGINT REFERENCES aulas(id) ON DELETE CASCADE,
  concluida BOOLEAN DEFAULT FALSE, melhor_pontuacao INTEGER DEFAULT 0,
  tentativas INTEGER DEFAULT 0, concluida_em TIMESTAMPTZ,
  UNIQUE(aluno_id, aula_id)
);
-- Badges
CREATE TABLE IF NOT EXISTS badges (
  id BIGSERIAL PRIMARY KEY, aluno_id BIGINT REFERENCES alunos(id) ON DELETE CASCADE,
  tipo TEXT NOT NULL, ganho_em TIMESTAMPTZ DEFAULT NOW()
);
-- RLS
ALTER TABLE aulas ENABLE ROW LEVEL SECURITY;
ALTER TABLE vocabulario ENABLE ROW LEVEL SECURITY;
ALTER TABLE frases ENABLE ROW LEVEL SECURITY;
ALTER TABLE alunos ENABLE ROW LEVEL SECURITY;
ALTER TABLE progresso ENABLE ROW LEVEL SECURITY;
ALTER TABLE badges ENABLE ROW LEVEL SECURITY;

Copie o SQL abaixo, vá em SQL Editor no painel do Supabase, cole e execute. Depois volte aqui. 
-- Tabela de aulas
CREATE TABLE IF NOT EXISTS aulas (
  id BIGSERIAL PRIMARY KEY,
  numero INTEGER NOT NULL,
  titulo TEXT NOT NULL,
  nivel TEXT DEFAULT 'A1',
  tema TEXT,
  verbos TEXT[],
  observacoes TEXT,
  criado_em TIMESTAMPTZ DEFAULT NOW()
);

-- Tabela de vocabulário
CREATE TABLE IF NOT EXISTS vocabulario (
  id BIGSERIAL PRIMARY KEY,
  aula_id BIGINT REFERENCES aulas(id) ON DELETE CASCADE,
  espanhol TEXT NOT NULL,
  portugues TEXT NOT NULL,
  categoria TEXT DEFAULT 'geral',
  exemplo TEXT,
  criado_em TIMESTAMPTZ DEFAULT NOW()
);

-- Tabela de frases
CREATE TABLE IF NOT EXISTS frases (
  id BIGSERIAL PRIMARY KEY,
  aula_id BIGINT REFERENCES aulas(id) ON DELETE CASCADE,
  espanhol TEXT NOT NULL,
  portugues TEXT NOT NULL,
  categoria TEXT DEFAULT 'geral',
  criado_em TIMESTAMPTZ DEFAULT NOW()
);

-- Permissões de leitura pública (para o app dos alunos)
ALTER TABLE aulas ENABLE ROW LEVEL SECURITY;
ALTER TABLE vocabulario ENABLE ROW LEVEL SECURITY;
ALTER TABLE frases ENABLE ROW LEVEL SECURITY;

CREATE POLICY "leitura publica aulas" ON aulas FOR SELECT USING (true);
CREATE POLICY "leitura publica vocab" ON vocabulario FOR SELECT USING (true);
CREATE POLICY "leitura publica frases" ON frases FOR SELECT USING (true);

CREATE POLICY "escrita autenticada aulas" ON aulas FOR ALL USING (true);
